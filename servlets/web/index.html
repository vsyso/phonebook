API Reference

GET /contact/{id}

PUT /contact/{id}

XML request example:

<?xml version="1.0" encoding="UTF-8"?>
<contact>
    <firstName>John</firstName>
    <lastName>Doe</lastName>
</contact>

JSON request example:

{"firstName":"John","lastName":"Doe"}

POST /contact

XML request example:

<?xml version="1.0" encoding="UTF-8"?>
<contact>
    <firstName>Jack</firstName>
    <lastName>Smith</lastName>
</contact>

JSON request example:

{"firstName":"Jack","lastName":"Smith"}

POST /contact/{id}/add_number

XML request example:

<?xml version="1.0" encoding="UTF-8"?>
<phoneNumber>
    <number>+38(056)734-34-22</number>
    <type>Home</type>
</phoneNumber>

JSON request example:

{"number":"+38(056)734-34-22","type":"Home"}

DELETE /contact/{id}

DELETE /contact/{id}/{phone_number}


GET /contacts

GET /contacts/find_by_number?{phone_number}
